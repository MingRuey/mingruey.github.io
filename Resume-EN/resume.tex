% create a coloured box with arrow and title as cv section headline
% param 1: section title
%
\newcommand{\cvsection}[2]
{
    \ifthenelse{\isempty{#1}}{}{
        \vspace{#1}
    }
	\begin{center}
		\large\textcolor{sectcol}{\textbf{#2}}
	\end{center}
}

%

{

%\end{tabular*}
}
